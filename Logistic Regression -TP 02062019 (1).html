def outlier(x):
    Q3 = x.describe()[6]
    Q1 = x.describe()[4]
    IQR = Q3-Q1
    upr = Q3 + 1.5*IQR
    lwr = Q1 - 1.5*IQR
    num_out = np.where((x < lwr) | (x > upr),1,0).sum()
    return(num_out,lwr,upr)
